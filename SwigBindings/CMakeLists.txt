
SET(SWIG_DIRS)

SET(BUILD_PYTHON_MOOS OFF CACHE BOOL "Use MOOS in python")
IF(BUILD_PYTHON_MOOS)
    SET(SWIG_DIRS ${SWIG_DIRS} python)
ENDIF(BUILD_PYTHON_MOOS)

SET(BUILD_JAVA_MOOS OFF CACHE BOOL "Use MOOS in Java")
IF(BUILD_JAVA_MOOS)
    SET(SWIG_DIRS ${SWIG_DIRS} java)
ENDIF(BUILD_JAVA_MOOS)

IF(SWIG_DIRS)
    SUBDIRS(${SWIG_DIRS})
ENDIF(SWIG_DIRS)
